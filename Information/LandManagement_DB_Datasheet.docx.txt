Entity · Attribute · Relation Datasheet  +  MySQL DDL

Document overview
This document covers the complete database design for the Land Management System, including 15 entities, all attributes with data types and constraints, entity-relationship table, and full MySQL DDL including triggers, events, indexes, and views.

Section 3 — MySQL DDL commands
Copy and execute the following commands in sequence in your MySQL client. Tables must be created in the order shown to satisfy foreign key dependencies.


— Database creation
CREATE DATABASE IF NOT EXISTS land_management_db
    CHARACTER SET utf8mb4
    COLLATE utf8mb4_unicode_ci;


USE land_management_db;
	

— ADMIN table
CREATE TABLE admin (
    admin_id       INT AUTO_INCREMENT PRIMARY KEY,
    full_name      VARCHAR(100) NOT NULL,
    email          VARCHAR(150) NOT NULL UNIQUE,
    password_hash  VARCHAR(255) NOT NULL,
    role           ENUM('superadmin','officer','surveyor') NOT NULL,
    last_login     DATETIME DEFAULT NULL,
    is_active      TINYINT(1) DEFAULT 1,
    created_at     DATETIME DEFAULT CURRENT_TIMESTAMP
);
	

— USER table
CREATE TABLE user (
    user_id        INT AUTO_INCREMENT PRIMARY KEY,
    full_name      VARCHAR(100) NOT NULL,
    nid_number     VARCHAR(30) NOT NULL UNIQUE,
    email          VARCHAR(150) DEFAULT NULL UNIQUE,
    phone          VARCHAR(20) NOT NULL,
    address        TEXT DEFAULT NULL,
    is_verified    TINYINT(1) DEFAULT 0,
    registered_at  DATETIME DEFAULT CURRENT_TIMESTAMP
);
	

— LAND table
CREATE TABLE land (
    land_id                  INT AUTO_INCREMENT PRIMARY KEY,
    parcel_number            VARCHAR(50) NOT NULL UNIQUE,
    area_sqft                DECIMAL(12,2) NOT NULL,
    land_type                ENUM('residential','agricultural',
                                  'commercial','industrial','government') NOT NULL,
    mouza                    VARCHAR(100) NOT NULL,
    upazila                  VARCHAR(100) NOT NULL,
    district                 VARCHAR(100) NOT NULL,
    current_status           ENUM('active','disputed','transferred','government')
                             DEFAULT 'active',
    current_estimated_value  DECIMAL(15,2) DEFAULT 0.00,
    registered_owner_id      INT NOT NULL,
    created_at               DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (registered_owner_id) REFERENCES user(user_id)
        ON UPDATE CASCADE ON DELETE RESTRICT
);
	

— LAND_TRANSACTION table
CREATE TABLE land_transaction (
    transaction_id    INT AUTO_INCREMENT PRIMARY KEY,
    land_id           INT NOT NULL,
    from_user_id      INT NOT NULL,
    to_user_id        INT NOT NULL,
    transaction_type  ENUM('sale','inheritance','gift','court_order') NOT NULL,
    transaction_date  DATE NOT NULL,
    amount            DECIMAL(15,2) DEFAULT 0.00,
    deed_number       VARCHAR(100) NOT NULL UNIQUE,
    chain_valid       TINYINT(1) DEFAULT 1,
    approved_by       INT DEFAULT NULL,
    created_at        DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (land_id)      REFERENCES land(land_id) ON DELETE RESTRICT,
    FOREIGN KEY (from_user_id) REFERENCES user(user_id) ON DELETE RESTRICT,
    FOREIGN KEY (to_user_id)   REFERENCES user(user_id) ON DELETE RESTRICT,
    FOREIGN KEY (approved_by)  REFERENCES admin(admin_id) ON DELETE SET NULL
);
	

— LAND_HISTORY table
CREATE TABLE land_history (
    history_id         INT AUTO_INCREMENT PRIMARY KEY,
    land_id            INT NOT NULL,
    event_type         ENUM('survey','transaction','dispute_filed',
                            'status_change','billboard_placed','work_order') NOT NULL,
    description        TEXT NOT NULL,
    event_date         DATE NOT NULL,
    recorded_by_admin  INT DEFAULT NULL,
    FOREIGN KEY (land_id)           REFERENCES land(land_id) ON DELETE RESTRICT,
    FOREIGN KEY (recorded_by_admin) REFERENCES admin(admin_id) ON DELETE SET NULL
);
	

— LAND_SURVEY table
CREATE TABLE land_survey (
    survey_id           INT AUTO_INCREMENT PRIMARY KEY,
    land_id             INT NOT NULL,
    survey_date         DATE NOT NULL,
    surveyor_admin_id   INT DEFAULT NULL,
    land_type_at_survey VARCHAR(50) NOT NULL,
    survey_method       ENUM('manual','digital','satellite') NOT NULL,
    notes               TEXT DEFAULT NULL,
    approved_by         INT DEFAULT NULL,
    FOREIGN KEY (land_id)           REFERENCES land(land_id) ON DELETE RESTRICT,
    FOREIGN KEY (surveyor_admin_id) REFERENCES admin(admin_id) ON DELETE SET NULL,
    FOREIGN KEY (approved_by)       REFERENCES admin(admin_id) ON DELETE SET NULL
);
	

— INFRASTRUCTURE table
CREATE TABLE infrastructure (
    asset_id          INT AUTO_INCREMENT PRIMARY KEY,
    land_id           INT NOT NULL,
    asset_name        VARCHAR(150) NOT NULL,
    asset_type        ENUM('building','road','drainage','bridge','utility') NOT NULL,
    construction_date DATE DEFAULT NULL,
    condition_status  ENUM('excellent','good','fair','poor','critical') DEFAULT 'good',
    owner_entity      VARCHAR(100) DEFAULT NULL,
    FOREIGN KEY (land_id) REFERENCES land(land_id) ON DELETE RESTRICT
);
	

— MAINTENANCE table
CREATE TABLE maintenance (
    maintenance_id     INT AUTO_INCREMENT PRIMARY KEY,
    asset_id           INT NOT NULL,
    maintenance_date   DATE NOT NULL,
    maintenance_type   ENUM('routine','emergency','upgrade','demolition') NOT NULL,
    cost               DECIMAL(12,2) DEFAULT 0.00,
    assigned_admin_id  INT DEFAULT NULL,
    notes              TEXT DEFAULT NULL,
    next_scheduled_date DATE DEFAULT NULL,
    FOREIGN KEY (asset_id)          REFERENCES infrastructure(asset_id) ON DELETE RESTRICT,
    FOREIGN KEY (assigned_admin_id) REFERENCES admin(admin_id) ON DELETE SET NULL
);
	

— WORK_ORDER table
CREATE TABLE work_order (
    work_order_id      INT AUTO_INCREMENT PRIMARY KEY,
    land_id            INT NOT NULL,
    work_type          ENUM('road_construction','survey','demolition',
                            'utility_install','inspection') NOT NULL,
    requested_by_user  INT DEFAULT NULL,
    assigned_admin     INT DEFAULT NULL,
    status             ENUM('pending','in_progress','completed','cancelled')
                       DEFAULT 'pending',
    start_date         DATE DEFAULT NULL,
    end_date           DATE DEFAULT NULL,
    notes              TEXT DEFAULT NULL,
    FOREIGN KEY (land_id)           REFERENCES land(land_id) ON DELETE RESTRICT,
    FOREIGN KEY (requested_by_user) REFERENCES user(user_id) ON DELETE SET NULL,
    FOREIGN KEY (assigned_admin)    REFERENCES admin(admin_id) ON DELETE SET NULL
);
	

— BILLBOARD table
CREATE TABLE billboard (
    billboard_id       INT AUTO_INCREMENT PRIMARY KEY,
    land_id            INT NOT NULL,
    owner_user_id      INT NOT NULL,
    license_number     VARCHAR(80) NOT NULL UNIQUE,
    size_sqft          DECIMAL(8,2) NOT NULL,
    installation_date  DATE NOT NULL,
    expiry_date        DATE NOT NULL,
    annual_fee         DECIMAL(10,2) NOT NULL,
    status             ENUM('active','expired','suspended','removed')
                       DEFAULT 'active',
    FOREIGN KEY (land_id)       REFERENCES land(land_id) ON DELETE RESTRICT,
    FOREIGN KEY (owner_user_id) REFERENCES user(user_id) ON DELETE RESTRICT
);
	

— BILLBOARD_PAYMENT table
CREATE TABLE billboard_payment (
    payment_id         INT AUTO_INCREMENT PRIMARY KEY,
    billboard_id       INT NOT NULL,
    payment_date       DATE NOT NULL,
    amount_paid        DECIMAL(10,2) NOT NULL,
    payment_method     ENUM('cash','bank_transfer','online') NOT NULL,
    status             ENUM('paid','pending','overdue') DEFAULT 'pending',
    received_by_admin  INT DEFAULT NULL,
    FOREIGN KEY (billboard_id)      REFERENCES billboard(billboard_id) ON DELETE RESTRICT,
    FOREIGN KEY (received_by_admin) REFERENCES admin(admin_id) ON DELETE SET NULL
);
	

— TAX table
CREATE TABLE tax (
    tax_id          INT AUTO_INCREMENT PRIMARY KEY,
    land_id         INT NOT NULL,
    owner_user_id   INT NOT NULL,
    financial_year  VARCHAR(10) NOT NULL,
    assessed_value  DECIMAL(15,2) NOT NULL,
    tax_rate        DECIMAL(5,4) NOT NULL,
    tax_amount      DECIMAL(12,2) NOT NULL,
    paid_amount     DECIMAL(12,2) DEFAULT 0.00,
    due_date        DATE NOT NULL,
    status          ENUM('unpaid','partial','paid','overdue') DEFAULT 'unpaid',
    calculated_by   INT DEFAULT NULL,

— Trigger — auto-escalate disputes (30-day idle)
DELIMITER $$


CREATE EVENT evt_escalate_disputes
ON SCHEDULE EVERY 1 DAY
DO
BEGIN
    UPDATE dispute
    SET
        escalation_level = LEAST(escalation_level + 1, 3),
        status = 'escalated'
    WHERE
        status NOT IN ('resolved','dismissed')
        AND escalation_level < 3
        AND DATEDIFF(NOW(), last_updated) > 30;
END$$


DELIMITER ;
	

— Trigger — auto-update document status
DELIMITER $$


CREATE EVENT evt_update_doc_status
ON SCHEDULE EVERY 1 DAY
DO
BEGIN
    UPDATE document_registration
    SET doc_status = 'expired'
    WHERE expiry_date < CURDATE()
      AND doc_status NOT IN ('expired','revoked');


    UPDATE document_registration
    SET doc_status = 'expiring_soon'
    WHERE expiry_date BETWEEN CURDATE()
                          AND DATE_ADD(CURDATE(), INTERVAL 30 DAY)
      AND doc_status = 'valid';
END$$


DELIMITER ;
	

— Trigger — update land estimated value after transaction
DELIMITER $$


CREATE TRIGGER trg_update_land_value
AFTER INSERT ON land_transaction
FOR EACH ROW
BEGIN
    UPDATE land
    SET current_estimated_value = (
        SELECT AVG(amount)
        FROM (
            SELECT amount
            FROM land_transaction
            WHERE land_id = NEW.land_id
              AND amount > 0
            ORDER BY transaction_date DESC
            LIMIT 3
        ) AS recent
    )
    WHERE land_id = NEW.land_id;
END$$


DELIMITER ;
	

— View — overdue taxes
CREATE VIEW vw_overdue_taxes AS
SELECT
    t.tax_id,
    l.parcel_number,
    l.district,
    u.full_name  AS owner_name,
    u.phone,
    t.financial_year,
    t.tax_amount,
    t.paid_amount,
    (t.tax_amount - t.paid_amount) AS balance_due,
    t.due_date
FROM tax t
JOIN land l ON t.land_id = l.land_id
JOIN user u ON t.owner_user_id = u.user_id
WHERE t.status IN ('unpaid','overdue','partial')
  AND t.due_date < CURDATE()
ORDER BY balance_due DESC;
	

— View — active dispute summary
CREATE VIEW vw_active_disputes AS
SELECT
    d.dispute_id,
    l.parcel_number,
    u.full_name        AS raised_by,
    d.dispute_type,
    d.status,
    d.escalation_level,
    d.filed_date,
    DATEDIFF(NOW(), d.last_updated) AS days_since_update,
    a.full_name        AS assigned_officer
FROM dispute d
JOIN land l  ON d.land_id = l.land_id
JOIN user u  ON d.raised_by_user = u.user_id
LEFT JOIN admin a ON d.handled_by_admin = a.admin_id
WHERE d.status NOT IN ('resolved','dismissed')
ORDER BY d.escalation_level DESC, d.filed_date ASC;
